An image component with placeholder functionality using the [SkeletonBlock](SkeletonBlock) component.

**Note**: Placeholder functionality is not supported in Internet Explorer (IE) 11. This is due to a bug in how IE11 handles the `onload` event for images. For all other browsers, the `Image` component must have an explicit height for the placeholder to appear.

### With useLoaderDuringTransitions

```jsx react-live=true
class StatefulExample extends React.Component {
    constructor() {
        this.state = {
            src: 'https://unsplash.it/400/300',
            useLoaderDuringTransitions: true
        }
    }

    transitionImage() {
        if (/300/.test(this.state.src)) {
            this.setState({src: 'https://unsplash.it/400/200'})
        } else {
            this.setState({src: 'https://unsplash.it/400/300'})
        }
    }

    changeUseLoaderDuringTransitions() {
        this.setState({useLoaderDuringTransitions: !this.state.useLoaderDuringTransitions})
    }

    render() {
        return (
            <span>
                <Image
                    src={this.state.src}
                    useLoaderDuringTransitions={this.state.useLoaderDuringTransitions}
                    artificialLoadingDelay={1000}
                    loadingIndicator={<img src={loading} />}
                />
                <button onClick={this.changeUseLoaderDuringTransitions.bind(this)}>
                    {' '}
                    {'useLoaderDuringTransitions=' + this.state.useLoaderDuringTransitions}
                </button>
                <button onClick={this.transitionImage.bind(this)}>
                    Change image src (simulates image load delay of 1000ms)
                </button>
            </span>
        )
    }
}
```

### With inline styles

Please note that the `width` property is overridden by the `imageStyle.width`.

```jsx react-live=true
class Example extends React.Component {
    constructor() {
        this.wrapperStyle = {
            position: 'relative',
            left: '160px'
        }

        this.imageStyle = {
            borderRadius: '10px',
            height: '80px',
            width: '80px'
        }

        this.cat =
            'https://s-media-cache-ak0.pinimg.com/564x/72/4b/6d/724b6d' +
            'bf91c378a53d6890bb525c1aa9.jpg'
    }

    render() {
        return (
            <Image
                src={this.cat}
                wrapperStyle={this.wrapperStyle}
                imageStyle={this.imageStyle}
                width="400px"
            />
        )
    }
}
```
